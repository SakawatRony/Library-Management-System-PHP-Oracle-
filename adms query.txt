    counter number;
    x number;
begin

for y in 1..50 loop   
    counter := 0;
    x := floor(y/2);
    for z in 2..x loop
        if (mod(y, z) = 0 ) then
            counter := 1;
        end if;
    end loop;

       dbms_output.put_line(y);
    end if;
  end loop;

exception
